cmake_minimum_required(VERSION 2.8.3)
project(srg-expressions)

add_compile_options(-std=c++11)

find_package(catkin REQUIRED alica_engine)

catkin_package(
  INCLUDE_DIRS autogenerated/include
  LIBRARIES srg-expressions
  CATKIN_DEPENDS alica_engine
)

include_directories(autogenerated/include ${catkin_INCLUDE_DIRS})

file(GLOB_RECURSE autogen_SOURCES "autogenerated/src/*.cpp")
add_library(srg-expressions ${autogen_SOURCES})

target_link_libraries(srg-expressions ${catkin_LIBRARIES})
