cmake_minimum_required(VERSION 2.8.3)
project(srg_expressions)

add_compile_options(-std=c++11)
set(CMAKE_BUILD_TYPE Debug)

find_package(catkin REQUIRED alica_engine capnzero system_config grid_sim srg_agent srg_worldmodel)

catkin_package(
  INCLUDE_DIRS autogenerated/include
  LIBRARIES srg_expressions
  CATKIN_DEPENDS alica_engine capnzero system_config grid_sim srg_agent srg_worldmodel
)

include_directories(autogenerated/include ${catkin_INCLUDE_DIRS})

file(GLOB_RECURSE autogen_SOURCES "autogenerated/src/*.cpp")
add_library(srg_expressions ${autogen_SOURCES} autogenerated/include/Behaviours/Stop.h autogenerated/include/Behaviours/Test.h autogenerated/include/Behaviours/Test1559635479527.h autogenerated/include/Behaviours/constraints/Stop1555602210283Constraints.h)

target_link_libraries(srg_expressions ${catkin_LIBRARIES})
